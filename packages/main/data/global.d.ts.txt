  function setImmediate(cb: (...args: any[]) => void, ...args: any[]): number;

  /** 应用上下文接口定义 */

    readonly path: string;
    readonly wid: number;
    getPath(partName: "meta" | "visualref"): string;
    notify(evtName: string, payload: unknown, srcId?: number): boolean;

  type Capability = {
    name: string;
    updatedAt: string;
    id: string;
    role: string;
    goal: string;
    code: string;
    dims: string[];
    input: string[];
    output: string[];
    process: string;
    negative: string;

    fewshot: {
      input: string[];

  /**维护本次工作流的状态，可以终止或向渲染进程发出消息。 */
  interface IRunnerContext {
    readonly signal: AbortSignal;
    readonly isAborted: boolean;
    readonly isForceKilled: boolean;
    readonly prj: IProjectContext;
    readonly cmd: {
      isCommand: boolean;

    push(cap: Capability): void;
    pop(): Capability | null;
    triggerAbort(force?: boolean): void;
    loadFunctor(capaId: string): ICapaFunctor | null;
    notify(message: string, ...args: unknown[]): void;
    info(message: string, ...args: unknown[]): void;
    warn(message: string, ...args: unknown[]): void;
    error(message: string, ...args: unknown[]): void;
    debug(message: string, ...args: unknown[]): void;
  }

  export interface ICapaFunctor {
    run(ctx: IRunnerContext): Promise<void>;

  interface PrjTimeStore<T> {

  const ctx: IRunnerContext;

  interface IValidator {
    isEmail(str: string): boolean;
    isMobilePhone(str: string, locale?: string): boolean;

  type IOInfo<IType = unknown, OType = unknown> = {
    expired: boolean; // 指示是否已过时，如果目标key不存在，或者目标key的更新时间小于input的时间，则设置为true.否则false.这个值指示是否需要重新计算output.
    inputs: Array<IType | null>;
    outputs: Array<OType | null>;
    inputsWithTime: Array<PrjTimeStore<IType> | null> | null;
    outputsWithTime: Array<PrjTimeStore<OType> | null> | null;
  };

  interface IGlossary {
    /**
     * 获取指定的IO数据，并判断是否失效。(资源，输入包括capa自身，更新时间都需要早于输出，否则expired.)
     * 但是未使用schema验证数据内容，策略是写入时验证，而不是读取时验证。
     * @param capa 能力对象
     * @param reses 依赖资源。 
     * @param inputKeys 输入metag数组，如未指定，使用capa.input.
     * @param outputKeys 输出metag数组，如未指定，使用capa.output.
     * @returns IOInfo
     */
    getIO<ITYpe = any, OType = any>(
      capa: Capability,
      reses?: string[],
      inputKeys?: string[],
      outputKeys?: string[],
    ): IOInfo<ITYpe, OType>;
    /**
     * 保存到metag的指定的位置，如果metag定义了schema,value必须能通过过其验证，否则失败。
     * @param outputKey metag的名称。 
     * @param value 
     * @returns 成功保存返回true,否则返回false。查看日志了解出错原因。
     */
    save(outputKey: string, value: unknown): boolean;

    /**
     * 直接获取原始数据(无更新时间).
     */
    get<T = string>(key: string): T | null;

    /**
     * 直接保存到原始数据(自动更新更新时间).
     */
    set(key: string, value: unknown, opt?: {
      reducer?: "merge" | "replace"
    }): void;
    /**
     * 验证数据是否符合schema定义。
     */
    validate: <T extends ZodType<any, any, any>>(schema: T,
      data: unknown
    ) => z.infer<T> | null;
  }
  const glossary: IGlossary;

  const err: {
    throwPrecondition(msg: string): never;
    throwNotfound(msg: string): never;
    throwNotimplement(msg: string): never;
    throwUnprcessable(msg: string): never;
    throwCancel(msg: string): never;
  };

  interface IRadashi {
    // 对象操作
    omit<T extends Record<string, any>>(obj: T, keys: string[]): Partial<T>;
    pick<T extends Record<string, any>>(obj: T, keys: string[]): Partial<T>;

    // 数组操作
    group<T>(array: T[], getKey: (item: T) => string): Record<string, T[]>;
    unique<T>(array: T[]): T[];

    // 函数式工具
    debounce<T extends (...args: any[]) => any>(
      fn: T,
      wait: number,
    ): (...args: Parameters<T>) => void;
    isString(str: unknown): string;
  }

  interface ZodType<Output = any, Def = any, Input = any> {
    readonly _output: Output;
    readonly _input: Input;
    readonly _def: Def;

    // 如果需要，可以顺便补全 zod 自带的 parse 方法
    parse(data: unknown): Output;
  }

  // 2. 模拟 z.infer 的类型提取器（通过读取 _output 属性拿到真实 TS 类型）
  namespace z {
    type infer<T extends ZodType<any, any, any>> = T extends ZodType<infer Output, any, any>
      ? Output
      : never;
  }

  const z: any;

  const util: IRadashi & {
    // 确保输入值转换为字符串
    toStrSafe(val: string | string[] | unknown): string;
    // 确保输入值转换为字符串数组
    toStrArraySafe(val: string | string[] | unknown): string[];
    /**
   * 能限制同时执行任务数、支持流式数据、且自带过滤功能的加强版 Promise.all。
   * @param input 包含 Promise 或普通值的可迭代对象（如数组）
   * @param mapper 异步映射回调函数
   * @param options 并发控制配置项
   */
    pMap: <Element, NewElement>(
      input: AsyncIterable<Element | Promise<Element>> | Iterable<Element | Promise<Element>>,
      mapper: (element: Element, index: number) => NewElement | Promise<NewElement>,
      options?: {
        readonly concurrency?: number;
        readonly stopOnError?: boolean;
        readonly signal?: AbortSignal;
      }
    ) => Promise<Array<NewElement>>;
  };
}
